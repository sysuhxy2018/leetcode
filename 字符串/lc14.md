class Solution {
    public String longestCommonPrefix(String[] strs) {
        int n = strs.length;
        int ind = 0;
        boolean end = false;
        
        StringBuilder sb = new StringBuilder();
        while (n > 0 && !end) {
            boolean same = true;
            for (int i = 0; i < n; i++) {
                if (ind >= strs[i].length()) {
                    end = true;
                    break;
                }
                if (i > 0 && strs[i].charAt(ind) != strs[i - 1].charAt(ind)) {
                    same = false;
                    break;
                }
            }
            if (end)
                break;
            if (same)
                sb.append(strs[0].charAt(ind));
            else
                break;  //No prefix any more, stop
            ind++;
        }
        
        return sb.toString();
    }
}
