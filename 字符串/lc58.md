class Solution {
    public int lengthOfLastWord(String s) {
        if (s.equals(""))
            return 0;
        char[] ss = s.toCharArray();
        int i = ss.length - 1;
        while (i >= 0) {
            if (ss[i] == ' ')
                i--;
            else {
                int cnt = 0;
                while (i >= 0 && ss[i] != ' ') {
                    cnt++;
                    i--;
                }
                return cnt;
            }
        }
        return 0;
    }
}
