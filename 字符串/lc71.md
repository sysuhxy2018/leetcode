class Solution {
    public String simplifyPath(String path) {
        if (path.equals(""))
            return "";
        char[] pc = path.toCharArray();
        Deque<String> st = new LinkedList<>();
        int i = 0;
        while (i < pc.length) {
            if (pc[i] == '/') {
                while (i < pc.length && pc[i] == '/')
                    i++;
            }
            else {
                StringBuilder sb = new StringBuilder();
                while (i < pc.length && pc[i] != '/') {
                    sb.append(pc[i]);
                    i++;
                }
                String dir = sb.toString();
                if (dir.equals("..")) {
                    if (!st.isEmpty())
                        st.pollLast();
                }
                else if (!dir.equals(".")) {
                    st.offerLast(dir);
                }
            }    
        }
        StringBuilder canon = new StringBuilder();
        canon.append('/');
        while (!st.isEmpty()) {
            canon.append(st.pollFirst());
            if (!st.isEmpty())
                canon.append('/');
        }
        return canon.toString();
    }
}
