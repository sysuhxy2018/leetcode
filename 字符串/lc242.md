class Solution {
    public boolean isAnagram(String s, String t) {
        char[] ss = s.toCharArray();
        char[] tt = t.toCharArray();
        if (ss.length != tt.length)
            return false;
        
        int[] letters = new int[26];
        for (char c : ss)
            letters[c - 'a']++;
        for (char c : tt)
            letters[c - 'a']--;
        for (int i = 0; i < 26; i++)
            if (letters[i] != 0)
                return false;
        return true;
    }
}
