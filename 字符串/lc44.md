class Solution {
    public boolean isMatch(String s, String p) {
        int m = s.length(), n = p.length();
        char[] ss = s.toCharArray();
        char[] pp = p.toCharArray();
        boolean[][] dp = new boolean[m + 1][n + 1];
        
        dp[0][0] = true;
        for (int i = 1; i <= n; i++)
            if (pp[i - 1] == '*')
                dp[0][i] = dp[0][i - 1];
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (pp[j - 1] == '?' || pp[j - 1] == ss[i - 1])
                    dp[i][j] = dp[i - 1][j - 1];
                else if (pp[j - 1] == '*') {
                    int k = i;
                    while (k >= 0) {
                        if (dp[k][j - 1]) {
                            dp[i][j] = true;
                            break;
                        }
                        k--;
                    }
                }
            }
        }
        
        return dp[m][n];
    }
}
