// 递归
class Solution {
    public TreeNode invertTree(TreeNode root) {
        if (root == null)
            return null;
        TreeNode a = invertTree(root.left);
        TreeNode b = invertTree(root.right);
        root.left = b;
        root.right = a;
        return root;
    } 
}

// 迭代
class Solution {
    public TreeNode invertTree(TreeNode root) {
        Queue<TreeNode> q = new LinkedList<>();
        if (root != null)
            q.offer(root);
        while (!q.isEmpty()) {
            int n = q.size();
            for (int i = 0; i < n; i++) {
                TreeNode tmp = q.poll();
                TreeNode a = tmp.left;
                TreeNode b = tmp.right;
                tmp.left = b;
                tmp.right = a;
                if (a != null)
                    q.offer(a);
                if (b != null)
                    q.offer(b);
            }
        }
        return root;
    }
}
