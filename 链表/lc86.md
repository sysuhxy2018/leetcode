class Solution {
    public ListNode partition(ListNode head, int x) {
        ListNode head1, tail1, head2, tail2, p;
        head1 = tail1 = new ListNode(-1);
        head2 = tail2 = new ListNode(-1);
        p = head;
        while (p != null) {
            if (p.val < x) {
                tail1.next = p;
                tail1 = p;
            }
            else {
                tail2.next = p;
                tail2 = p;
            }
            p = p.next;
        }
        tail1.next = head2.next;
        tail2.next = null;
        return head1.next;
    }
}
