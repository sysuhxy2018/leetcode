class Solution {
    public void reorderList(ListNode head) {
        
        if (head == null || head.next == null)
            return;
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        
        ListNode fast, slow, prev;
        fast = slow = head;
        prev = dummy;
        
        while (fast != null && fast.next != null) {
            prev = slow;
            slow = slow.next;
            fast = fast.next.next;
        }
        
        prev.next = null;
        ListNode head2 = null;
        while (slow != null) {
            ListNode next = slow.next;
            slow.next = head2;
            head2 = slow;
            slow = next;
        }
        
        fast = head;
        slow = head2;
        ListNode h, t;
        h = t = new ListNode(0);
        
        while (fast != null || slow != null) {
            if (fast == null) {
                t.next = slow;
                break;
            }
            else if (slow == null) {
                t.next = fast;
                break;
            }
            else {
                ListNode fNext = fast.next, sNext = slow.next;  // 坑点
                t.next = fast;
                t = fast;
                t.next = slow;
                t = slow;
                
                fast = fNext;
                slow = sNext;
            }
        }
        
        //head = h.next;
    }
}
