``` java
class Solution {
    public ListNode removeNthFromEnd(ListNode head, int n) {
        if (head == null)
            return head;
        ListNode p1, p2, h, t;
        p1 = p2 = head;
        h = t = new ListNode(0);
        for (int i = 0; i < n; i++)
            p1 = p1.next;
        while (p1 != null) {
            t.next = p2;
            t = p2;
            p1 = p1.next;
            p2 = p2.next;
        }
        t.next = p2.next;
        p2.next = null;
        return h.next;
    }
}
```
