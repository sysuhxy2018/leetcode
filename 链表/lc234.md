class Solution {
    public boolean isPalindrome(ListNode head) {
        ListNode fast, slow;
        fast = slow = head;
        while (fast != null && fast.next != null) {
            fast = fast.next.next;
            slow = slow.next;
        }
        
        ListNode h = null;
        while (slow != null) {
            ListNode next = slow.next;
            slow.next = h;
            h = slow;
            slow = next;
        }
        
        fast = head;
        slow = h;
        while (slow != null && fast != null) {
            if (slow.val != fast.val)
                return false;
            slow = slow.next;
            fast = fast.next;
        }
        return true;
    }
}
