``` java
/*
// Definition for a Node.
class Node {
    public int val;
    public Node next;
    public Node random;

    public Node() {}

    public Node(int _val,Node _next,Node _random) {
        val = _val;
        next = _next;
        random = _random;
    }
};
*/
class Solution {
    public Node copyRandomList(Node head) {
        Node p = head;
        while (p != null) {
            Node node = new Node();
            node.val = p.val;
            
            Node next = p.next;
            p.next = node;
            node.next = next;
            p = next;
        }
        
        p = head;
        while (p != null) {
            if (p.random != null) {
                Node nRandom = p.random.next;
                p.next.random = nRandom;
            }
            p = p.next.next;
        }
        
        Node h, t;
        h = t = new Node();
        p = head;
        while (p != null) {
            t.next = p.next;
            t = p.next;
            
            p.next = null;
            p = t.next;
        }
        t.next = null;
        return h.next;
    }
}
```
