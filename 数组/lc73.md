import java.util.*;

// 空间复杂度O(m + n)
class Solution {
    public void setZeroes(int[][] matrix) {
        int m = matrix.length;
        if (m <= 0)
            return;
        int n = matrix[0].length;
        BitSet a = new BitSet(m);
        BitSet b = new BitSet(n);
        for (int i = 0; i < m; i++)
            for (int j = 0; j < n; j++)
                if (matrix[i][j] == 0) {
                    a.set(i); b.set(j);
                }
        for (int i = 0; i < m; i++)
            if (a.get(i))
                for (int j = 0; j < n; j++)
                    matrix[i][j] = 0;
        for (int i = 0; i < n; i++)
            if (b.get(i))
                for (int j = 0; j < m; j++)
                    matrix[j][i] = 0;      
    }
}

// 空间复杂度O(1)
class Solution {
    public void setZeroes(int[][] matrix) {
        int m = matrix.length;
        if (m <= 0)
            return;
        int n = matrix[0].length;
        boolean firstRow = false, firstCol = false;
        for (int i = 0; i < n; i++)
            if (matrix[0][i] == 0)
                firstRow = true;
        for (int i = 0; i < m; i++)
            if (matrix[i][0] == 0)
                firstCol = true;
        for (int i = 1; i < m; i++)
            for (int j = 1; j < n; j++)
                if (matrix[i][j] == 0) {
                    matrix[i][0] = 0;
                    matrix[0][j] = 0;
                }
        for (int i = 1; i < m; i++)
            if (matrix[i][0] == 0)
                for (int j = 1; j < n; j++)
                    matrix[i][j] = 0;
        for (int i = 1; i < n; i++)
            if (matrix[0][i] == 0)
                for (int j = 1; j < m; j++)
                    matrix[j][i] = 0;
        if (firstRow)
            for (int i = 0; i < n; i++)
                matrix[0][i] = 0;
        if (firstCol)
            for (int i = 0; i < m; i++)
                matrix[i][0] = 0;
    }
}
