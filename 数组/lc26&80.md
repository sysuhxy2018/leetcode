### 代码

``` java
class Solution {
    private static final int k = 2;
    public int removeDuplicates(int[] nums) {
        int n = nums.length;
        int i = 0, j = 0;
        while (i < n) {
            if (i + 1 < n && nums[i] == nums[i + 1]) {
                int cnt = 0;
                while (i + 1 < n && nums[i] == nums[i + 1]) {
                    if (cnt < k) {
                        nums[j++] = nums[i];
                        cnt++;
                    }
                    i++;
                }
                if (cnt < k) {
                    nums[j++] = nums[i];
                    cnt++;
                }
            }
            else {
                nums[j++] = nums[i];
            }
            i++;
        }
        return j;
    }
}
```
